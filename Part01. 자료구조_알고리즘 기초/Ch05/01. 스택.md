class Stack {
    constructor(){
        this.top = null;
        this.size = 0;
    }

    // Push
    push(value) {
        const node = new Node(value);
        node.next = this.top;
        this.top = node;
        this.size += 1;
    }

    // Pop
    pop() {
        const value = this.top.value;
        this.top = this.top.next;
        this.size -= 1;
        return value;
    }

    size() {
        return this.size;
    }

}

const stack = new Stack();
